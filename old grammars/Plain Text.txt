Cats is the inflected version of the lexeme cat, which is of type common-noun-lex 
and that inherits the following constraint from supertype obl-spr-noun-lex
[ SYNSEM.LOCAL.CAT.VAL.SPR < [ OPT - ] > ].

Bare NP rule states that head daughter must be of the feature [ OPT + ]

HEAD-DTR.SYNSEM.LOCAL [ CAT [ VAL [ SPR < [ OPT + ] >] ] ]

no-spr-noun-lex := noun-lex &
  [ SYNSEM.LOCAL.CAT.VAL.SPR < [ OPT + ] > ].

basic-head-spec-phrase-super := binary-nonloc-phrase & phrasal &
              binary-headed-phrase &
  [ INFLECTED infl-satisfied,
    SYNSEM [ LOCAL.CAT [ WH [ OR < #or1, #or2 > ],
                         VAL [ SUBJ #subj,
                               COMPS #spcomps,
                               SPR #spr,
                               SPEC #spec ],
                         POSTHEAD #ph,
                         HC-LIGHT - ],
              MODIFIED #modif ],
    HEAD-DTR [ INFLECTED infl-satisfied,
               SYNSEM [ LOCAL [ CAT [ WH #or1,
                                      HEAD #head,
                                      VAL [ SUBJ #subj,
                                            COMPS olist & #comps,
                                            SPR < #synsem &
                                                  canonical-synsem . #spr >,
                                            SPEC #spec ],
                                      POSTHEAD #ph ],
                                CONT.HOOK #hdhook ],
                        MODIFIED #hmodif ] ],
    NON-HEAD-DTR.SYNSEM #synsem &
     [ LOCAL [ CAT [ WH #or2, VAL [ SPEC < [ LOCAL [ CAT [ HEAD #head,
                                                  VAL.COMPS #comps ],
                                            CONT.HOOK #hdhook ],
                                    MODIFIED #hmodif ] >,
                           COMPS #spcomps ] ] ],
       MODIFIED #modif ],
    C-CONT [ RELS.LIST < >, HCONS.LIST < >, ICONS.LIST < > ] ].
